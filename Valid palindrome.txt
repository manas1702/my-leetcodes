class Solution {
public:
    bool isPalindrome(string s) {
        int start=0, end = s.size()-1;
        if(s.length()<=1) return true;

        while(start<=end)
        {
            if(!isalnum(s[start])){
                start++;
                continue;
            } 
            if(!isalnum(s[end])){
                end--;
                continue;
            } 
            if(tolower(s[start])!=tolower(s[end])) 
            {
                return false;
            }
           
                start++;
                end--;
            
        }
        return true;
    }
};
